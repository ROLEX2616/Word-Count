## PROGRAM:

```

# Word count in a Text file
# Developed by: KAVINRAJ.S
# Register number: 212223100019
num=0
with open ("sam.txt","r") as f1:
    for i in f1:
        word=i.split()
        num += len(word)
print("The number of words in the file is",num)
```
